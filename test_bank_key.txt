Q05,Medium,Calculate the average review score specifically for product items translated to the english category 'Health_beauty'.
Note: the search value ‘health_beauty’ is deliberately mentioned as  'Health_beauty' so that sql handles case inconsistencies.

FROM order_reviews r
JOIN orders o ON r.order_id = o.order_id
JOIN order_items i ON o.order_id = i.order_id
JOIN products p ON i.product_id = p.product_id

Q08,High,Show the month-over-month growth rate in total order count for the first half of 2018.

WITH
monthly_orders AS (
SELECT
DATE_TRUNC('month', order_purchase_timestamp) AS order_month,
COUNT(order_id) AS current_month_orders
FROM orders
WHERE order_purchase_timestamp >= '2017-12-01 00:00:00'
AND order_purchase_timestamp <= '2018-06-30 23:59:59'
GROUP BY 1),
lag_orders AS (
SELECT
order_month,
current_month_orders,
LAG(current_month_orders) OVER (ORDER BY order_month) AS prev_month_orders
FROM monthly_orders)
SELECT order_month, current_month_orders, prev_month_orders,
ROUND(((current_month_orders - prev_month_orders) * 100.0 / prev_month_orders), 2) AS mom_growth_percent
FROM lag_orders
where year(order_month)=2018;

Q14,High,Find the total late deliveries and their average customer review score grouped by seller state showing only states with more than 100 late deliveries.

SELECT
    s.seller_state,
    COUNT(DISTINCT o.order_id) as total_late_deliveries,
    AVG(r.review_score) as avg_review_score
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
JOIN sellers s ON oi.seller_id = s.seller_id
LEFT JOIN order_reviews r ON o.order_id = r.order_id
WHERE o.order_delivered_customer_date > o.order_estimated_delivery_date
GROUP BY s.seller_state
HAVING COUNT(DISTINCT o.order_id) > 100
ORDER BY total_late_deliveries DESC

Q15,High,Analyze customer retention by calculating the total GMV contribution, average GMV contribution and overall average review score of repeat customers vs single-purchase customers.


WITH customer_order_counts AS (
  SELECT
    c.customer_unique_id,
    COUNT(DISTINCT o.order_id) as order_count
  FROM customers c
  JOIN orders o ON c.customer_id = o.customer_id
  GROUP BY c.customer_unique_id
),
customer_segments AS (
  SELECT
    customer_unique_id,
    CASE
      WHEN order_count = 1 THEN 'Single Purchase'
      ELSE 'Repeat Customer'
    END as customer_type
  FROM customer_order_counts
),
gmv_by_segment AS (
  SELECT
    cs.customer_type,
    SUM(oi.price + oi.freight_value) as total_gmv,
    COUNT(DISTINCT c.customer_unique_id) as customer_count
  FROM customer_segments cs
  LEFT JOIN customers c ON cs.customer_unique_id = c.customer_unique_id
  LEFT JOIN orders o ON c.customer_id = o.customer_id
  LEFT JOIN order_items oi ON o.order_id = oi.order_id
  GROUP BY cs.customer_type
),
reviews_by_segment AS (
  SELECT
    cs.customer_type,
    AVG(r.review_score) as avg_review_score
  FROM customer_segments cs
  LEFT JOIN customers c ON cs.customer_unique_id = c.customer_unique_id
  LEFT JOIN orders o ON c.customer_id = o.customer_id
  LEFT JOIN order_reviews r ON o.order_id = r.order_id
  GROUP BY cs.customer_type
)
SELECT
  g.customer_type,
  g.customer_count,
  g.total_gmv,
  ROUND(g.total_gmv / g.customer_count, 2) as avg_gmv_per_customer,
  ROUND(r.avg_review_score, 2) as avg_review_score
FROM gmv_by_segment g
JOIN reviews_by_segment r ON g.customer_type = r.customer_type
ORDER BY g.customer_type;
